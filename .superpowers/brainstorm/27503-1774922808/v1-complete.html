Quick Start

# Install Rust (if needed)
curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh

# Install dependencies
cd neureaux-devtools && pnpm install

# Build cloudmock gateway
cd ../cloudmock && make build-gateway

# Run the app
cd ../neureaux-devtools && pnpm tauri dev
